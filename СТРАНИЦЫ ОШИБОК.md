// ========== ФУНКЦИЯ СОЗДАНИЯ СТРАНИЦЫ ==========
function createErrorPage(pageName, config) {
    let additionalHtml = '';
    let additionalScripts = '';
    
    // Добавляем дополнительные секции
    if (config.hasTimer) {
        additionalHtml += `
        <div class="timer">
            <div class="timer-value" id="countdown">60</div>
            <div class="timer-label">секунд до восстановления доступа</div>
        </div>`;
        
        additionalScripts += `
        <script>
            let timeLeft = 60;
            const timerElement = document.getElementById('countdown');
            const retryBtn = document.getElementById('retryBtn');
            
            const countdown = setInterval(() => {
                timeLeft--;
                timerElement.textContent = timeLeft;
                
                if (timeLeft <= 0) {
                    clearInterval(countdown);
                    timerElement.textContent = '0';
                    if (retryBtn) {
                        retryBtn.disabled = false;
                        retryBtn.innerHTML = '<i class="fas fa-redo-alt"></i> Попробовать снова';
                        retryBtn.onclick = () => window.location.reload();
                    }
                }
            }, 1000);
        </script>`;
    }
    
    if (config.hasReport) {
        additionalHtml += `
        <div class="error-report">
            <p><i class="fas fa-bug"></i> Сообщить об ошибке:</p>
            <textarea id="errorReport" rows="3" placeholder="Опишите, что вы делали перед возникновением ошибки..."></textarea>
            <button class="btn btn-primary report-btn" id="reportBtn" style="margin-top: 10px; width: 100%;">
                <i class="fas fa-paper-plane"></i> Отправить отчет
            </button>
        </div>`;
        
        additionalScripts += `
        <script>
            document.getElementById('reportBtn')?.addEventListener('click', () => {
                const report = document.getElementById('errorReport').value;
                if (report) {
                    alert('Спасибо за сообщение! Мы исправим проблему в ближайшее время.');
                    document.getElementById('errorReport').value = '';
                } else {
                    alert('Пожалуйста, опишите проблему перед отправкой.');
                }
            });
        </script>`;
    }
    
    if (config.hasMaintenance) {
        additionalHtml += `
        <div class="maintenance-notice">
            <i class="fas fa-clock"></i> Ориентировочное время восстановления: 15-30 минут
        </div>`;
    }
    
    if (config.hasAutoRefresh) {
        additionalHtml += `
        <div class="timer">
            <div class="timer-value" id="countdown">10</div>
            <div class="timer-label">автоматическое обновление через</div>
        </div>`;
        
        additionalScripts += `
        <script>
            let timeLeft = 10;
            const timerElement = document.getElementById('countdown');
            
            const countdown = setInterval(() => {
                timeLeft--;
                timerElement.textContent = timeLeft;
                
                if (timeLeft <= 0) {
                    clearInterval(countdown);
                    location.reload();
                }
            }, 1000);
        </script>`;
    }
    
    // Создаем список причин
    let detailsHtml = '';
    if (config.details && config.details.length > 0) {
        detailsHtml = `
        <div class="error-details">
            <p><i class="fas fa-info-circle"></i> Возможные причины:</p>
            ${config.details.map(detail => `<p><i class="fas fa-angle-right"></i> ${detail}</p>`).join('')}
        </div>`;
    }
    
    // Анимация
    let animationStyle = '';
    if (config.animation) {
        if (config.animation === 'shake') {
            animationStyle = `
            @keyframes shake {
                0%, 100% { transform: translateX(0); }
                25% { transform: translateX(-10px); }
                75% { transform: translateX(10px); }
            }
            .error-icon { animation: shake 0.5s ease; }`;
        } else if (config.animation === 'pulse') {
            animationStyle = `
            @keyframes pulse {
                0%, 100% { transform: scale(1); }
                50% { transform: scale(1.1); }
            }
            .error-icon { animation: pulse 1s ease; }`;
        } else if (config.animation === 'swing') {
            animationStyle = `
            @keyframes swing {
                0%, 100% { transform: rotate(0deg); }
                25% { transform: rotate(-15deg); }
                75% { transform: rotate(15deg); }
            }
            .error-icon { animation: swing 0.5s ease; }`;
        } else if (config.animation === 'float') {
            animationStyle = `
            @keyframes float {
                0%, 100% { transform: translateY(0); }
                50% { transform: translateY(-10px); }
            }
            .error-icon { animation: float 2s ease-in-out infinite; }`;
        }
    }
    
    // Полный HTML
    const html = `<!DOCTYPE html>
<html lang="ru">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>${config.code} - ${config.title} | Modern Messenger</title>
    <link href="https://fonts.googleapis.com/css2?family=Inter:wght@300;400;500;600;700&display=swap" rel="stylesheet">
    <link rel="stylesheet" href="https://cdnjs.cloudflare.com/ajax/libs/font-awesome/6.0.0-beta3/css/all.min.css">
    <style>
        ${baseStyles}
        ${animationStyle}
        
        .error-details {
            background: #f8f9fa;
            border-radius: 15px;
            padding: 20px;
            margin-bottom: 30px;
            text-align: left;
        }
        
        .error-details p {
            margin: 10px 0;
            color: #555;
        }
        
        .error-details i {
            color: ${config.iconColor};
            width: 25px;
        }
        
        .timer {
            background: #f8f9fa;
            padding: 20px;
            border-radius: 15px;
            margin-bottom: 30px;
        }
        
        .timer-value {
            font-size: 48px;
            font-weight: 700;
            color: #667eea;
        }
        
        .timer-label {
            font-size: 14px;
            color: #666;
        }
        
        .error-report {
            background: #f8f9fa;
            border-radius: 15px;
            padding: 20px;
            margin-bottom: 30px;
            text-align: left;
        }
        
        .error-report textarea {
            width: 100%;
            padding: 10px;
            border: 2px solid #e0e0e0;
            border-radius: 10px;
            font-size: 14px;
            font-family: inherit;
            resize: vertical;
            margin-top: 10px;
        }
        
        .maintenance-notice {
            background: linear-gradient(135deg, #667eea, #764ba2);
            color: white;
            padding: 20px;
            border-radius: 15px;
            margin-bottom: 30px;
        }
        
        .report-btn {
            margin-top: 10px;
            width: 100%;
        }
    </style>
</head>
<body>
    <div class="error-container">
        <div class="error-icon" style="color: ${config.iconColor}">
            <i class="fas ${config.icon}"></i>
        </div>
        <div class="error-code">${config.code}</div>
        <h1 class="error-title">${config.title}</h1>
        <p class="error-description">${config.description}</p>
        ${detailsHtml}
        ${additionalHtml}
        <div class="btn-group">
            <a href="/" class="btn btn-primary"><i class="fas fa-home"></i> На главную</a>
            <a href="javascript:history.back()" class="btn btn-secondary"><i class="fas fa-arrow-left"></i> Назад</a>
            ${config.code === '429' ? '<button class="btn btn-primary" id="retryBtn" disabled><i class="fas fa-redo-alt"></i> Попробовать снова</button>' : ''}
        </div>
    </div>
    ${additionalScripts}
</body>
</html>`;
    
    const filePath = path.join(errorsDir, pageName);
    fs.writeFileSync(filePath, html, 'utf8');
    console.log(`✅ Создана страница: errors/${pageName}`);
}

// ========== СОЗДАНИЕ ВСЕХ СТРАНИЦ ==========
console.log('📝 Создание страниц ошибок...\n');

Object.entries(errorPages).forEach(([fileName, config]) => {
    createErrorPage(fileName, config);
});
